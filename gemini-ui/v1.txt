                {tab === 'ALL' ? '全部任务' : tab === 'ACTIVE' ? '活跃作战' : '已归档案'}
              </button>
            ))}
          </div>

          <div className="relative w-full sm:w-72 group">
            <Search className={`absolute left-3 top-1/2 -translate-y-1/2 w-4 h-4 transition-colors ${isRei ? 'text-blue-400 group-focus-within:text-cyan-500' : 'text-purple-500 group-focus-within:text-green-400'}`} />
            <input 
              type="text" 
              placeholder="检索任务代号或名称..." 
              value={searchQuery}
              onChange={(e) => setSearchQuery(e.target.value)}
              className={`w-full pl-9 pr-4 py-2 rounded-sm outline-none font-mono text-sm transition-all border ${theme.inputBg}`}
            />
          </div>
        </div>

        <div className="flex-1 overflow-y-auto custom-scrollbar pr-2 pb-8">
          <div className="grid grid-cols-1 lg:grid-cols-3 gap-5">
            {filteredTasks.map(task => (
              <div 
                key={task.id} 
                onClick={() => { setSelectedTask(task.id); setProjectView('workbench'); }}
                className={`group relative backdrop-blur-md border p-5 rounded-sm cursor-pointer transition-all duration-300 flex flex-col min-h-[260px] h-auto ${theme.cardBg}`}
              >
                <div className="flex justify-between items-center mb-3">
                  {getStatusBadge(task.status)}
                  <div className={`text-xs font-mono tracking-widest opacity-60 ${theme.cardSub}`}>{task.id}</div>
                </div>

                <div className="flex-1">
                  <h3 className={`text-lg font-bold mb-2 tracking-wide ${theme.cardTitle}`}>{task.title}</h3>
                  <p className={`text-sm leading-relaxed ${theme.cardSub}`}>
                    {task.description}
                  </p>
                </div>

                <div className={`mt-4 p-2.5 rounded-sm border font-mono text-xs flex items-center justify-between ${theme.pathBg}`}>
                  <div className="flex items-center overflow-hidden">
                    <GitBranch className={`w-3.5 h-3.5 mr-2 shrink-0 ${theme.pathLabel}`} />
                    <span className="truncate max-w-[80px]" title={task.baseBranch}>{task.baseBranch}</span>
                    <ChevronRight className={`w-3 h-3 mx-1 shrink-0 ${isRei ? 'text-blue-300' : 'text-purple-700'}`} />
                    <span className={`font-bold truncate ${isRei ? 'text-cyan-600' : 'text-green-400'}`} title={task.taskBranch}>{task.taskBranch}</span>
                  </div>
                </div>

                <div className={`flex items-center justify-between mt-4 pt-4 border-t ${isRei ? 'border-blue-100/50' : 'border-white/5'}`}>
                  <div className="flex-1 mr-4">
                     <div className={`h-1 w-full rounded-full overflow-hidden ${isRei ? 'bg-blue-100' : 'bg-white/10'}`}>
                        <div 
                          className={`h-full transition-all duration-1000 ${isRei ? 'bg-cyan-400' : 'bg-green-400'}`} 
                          style={{ width: `${task.progress}%` }}
                        ></div>
                     </div>
                  </div>

                  <div className="flex items-center space-x-1 shrink-0">
                    <button 
                      onClick={(e) => openTaskModal('pause', task, e)}
                      className={`p-1.5 rounded-sm transition-colors ${theme.btnGhost}`}
                      title={task.status === 'PAUSED' ? '恢复执行' : '暂停任务'}
                    >
                      {task.status === 'PAUSED' ? <PlayCircle className="w-4 h-4" /> : <PauseCircle className="w-4 h-4" />}
                    </button>
                    <button 
                      onClick={(e) => openTaskModal('archive', task, e)}
                      className={`p-1.5 rounded-sm transition-colors ${theme.btnGhost}`}
                      title="归档至冷数据区"
                    >
                      <Archive className="w-4 h-4" />
                    </button>
                    <button 
                      onClick={(e) => openTaskModal('delete', task, e)}
                      className={`p-1.5 rounded-sm transition-colors ${theme.btnDanger}`}
                      title="彻底抹除记录"
                    >
                      <Trash2 className="w-4 h-4" />
                    </button>
                  </div>
                </div>

                <div className={`absolute top-0 left-0 w-2 h-2 border-t-2 border-l-2 transition-colors ${theme.cardCorner}`}></div>
                <div className={`absolute bottom-0 right-0 w-2 h-2 border-b-2 border-r-2 transition-colors ${theme.cardCorner}`}></div>
              </div>
            ))}
          </div>
          
          {filteredTasks.length === 0 && (
            <div className={`flex flex-col items-center justify-center py-20 font-mono ${theme.cardSub}`}>
              <AlertTriangle className={`w-8 h-8 mb-4 opacity-50 ${isRei ? 'text-blue-400' : 'text-purple-500'}`} />
              <div>未检索到符合条件的作战任务。</div>
            </div>
          )}
        </div>
      </div>
    );
  };

  const PlaceholderPage = ({ title, desc }) => (
    <div className="flex flex-col items-center justify-center h-[calc(100vh-4rem)] relative z-10">
      <div className="w-24 h-24 mb-6 relative">
        <div className={`absolute inset-0 border-2 animate-[spin_10s_linear_infinite] rounded-full ${theme.placeholderIcon}`}></div>
        <div className={`absolute inset-2 border-2 border-dashed animate-[spin_7s_linear_infinite_reverse] rounded-full ${isRei ? 'border-cyan-400/30' : 'border-green-400/20'}`}></div>
        <div className="absolute inset-0 flex items-center justify-center">
          <AlertTriangle className={`w-8 h-8 opacity-50 ${isRei ? 'text-blue-500' : 'text-purple-400'}`} />
        </div>
      </div>
      <h2 className={`text-2xl font-bold mb-2 font-mono tracking-widest ${isRei ? 'text-slate-800' : 'text-slate-200'}`}>[{title}]</h2>
      <p className="text-slate-500 font-mono text-sm">{desc}</p>
    </div>
  );

  return (
    <div className={`flex h-screen w-full overflow-hidden font-sans relative transition-colors duration-500 ${theme.bg}`}>
      {/* 网格背景 */}
      <div className="absolute inset-0 pointer-events-none opacity-20 transition-all duration-500" style={{ backgroundImage: theme.grid, backgroundSize: '40px 40px' }}></div>
      {/* 模糊色块 */}
      <div className={`absolute -top-[20%] -left-[10%] w-[50vw] h-[50vw] blur-[120px] rounded-full pointer-events-none transition-colors duration-1000 ${theme.blur1}`}></div>
      <div className={`absolute -bottom-[20%] -right-[10%] w-[40vw] h-[40vw] blur-[100px] rounded-full pointer-events-none transition-colors duration-1000 ${theme.blur2}`}></div>
      <div className={`absolute top-[30%] right-[20%] w-[20vw] h-[20vw] blur-[80px] rounded-full pointer-events-none transition-colors duration-1000 ${theme.blur3}`}></div>
      {/* 扫描线特效 */}
      <div className="absolute inset-0 pointer-events-none bg-[url('data:image/svg+xml;base64,PHN2ZyB4bWxucz0iaHR0cDovL3d3dy53My5vcmcvMjAwMC9zdmciIHdpZHRoPSI0IiBoZWlnaHQ9IjQiPjxyZWN0IHdpZHRoPSI0IiBoZWlnaHQ9IjEiIGZpbGw9InJnYmEoMCwwLDAsMC4yKSIvPjwvc3ZnPg==')] opacity-50 mix-blend-overlay"></div>

      {/* 弹窗：注册新项目文件树 */}
      <RegisterProjectModal />

      {/* 弹窗：项目移除拦截与确认 */}
      <DeleteProjectModal />
      
      {/* 弹窗：任务操作（拦截/确认） */}
      <TaskActionModal />

      <Sidebar />
      <div className="flex-1 flex flex-col min-w-0 relative z-10">
        <Header />
        <main className="flex-1 overflow-hidden relative">
          {!selectedProject && activeMenu === 'console' && <PlaceholderPage title="系统控制台" desc="等待神经连接... 全局监控指标离线。" />}
          {!selectedProject && activeMenu === 'projects' && <ProjectListPage />}
          {!selectedProject && activeMenu === 'settings' && <PlaceholderPage title="系统全局配置" desc="MAGI 系统配置接口已锁定，需解除权限限制。" />}
          {selectedProject && projectView === 'taskCenter' && <TaskCenterPage />}
          {selectedProject && projectView === 'workbench' && <PlaceholderPage title="核心工作台" desc="多智能体特遣系统已初始化。正在与本地环境同步..." />}
        </main>
      </div>
    </div>
  );
}
