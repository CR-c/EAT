// --- 递归文件树节点组件 ---
const FileTreeNode = ({ node, level, selectedDiffFile, setSelectedDiffFile, theme, isRei }) => {
  const [isOpen, setIsOpen] = useState(true);
  const isFile = node.type === 'file';
  const isSelected = isFile && selectedDiffFile?.path === node.path;

  if (node.name === 'root') {
    return (
      <div className="py-2">
        {Object.values(node.children || {}).map(child => (
          <FileTreeNode key={child.path} node={child} level={0} selectedDiffFile={selectedDiffFile} setSelectedDiffFile={setSelectedDiffFile} theme={theme} isRei={isRei} />
        ))}
      </div>
    );
  }

  return (
    <div className="flex flex-col select-none">
      <div
        className={`flex items-center justify-between py-1.5 pr-3 cursor-pointer transition-colors ${isSelected ? (isRei ? 'bg-blue-100/60 border-r-2 border-blue-500' : 'bg-purple-900/40 border-r-2 border-purple-500') : (isRei ? 'hover:bg-slate-100/50' : 'hover:bg-white/5')}`}
        style={{ paddingLeft: `${level * 16 + 12}px` }}
        onClick={() => {
          if (isFile) setSelectedDiffFile(node.fileData);
          else setIsOpen(!isOpen);
        }}
      >
        <div className="flex items-center overflow-hidden">
          {!isFile ? (
            <div className="flex items-center">
               <ChevronRight className={`w-3 h-3 mr-1 transition-transform ${isOpen ? 'rotate-90' : ''} ${theme.cardSub}`} />
               {isOpen ? <FolderOpen className={`w-4 h-4 mr-2 shrink-0 ${isRei ? 'text-blue-500' : 'text-purple-400'}`} /> : <Folder className={`w-4 h-4 mr-2 shrink-0 ${isRei ? 'text-blue-500' : 'text-purple-400'}`} />}
            </div>
          ) : (
            <div className="flex items-center pl-4">
               <FileText className={`w-4 h-4 mr-2 shrink-0 ${isSelected ? (isRei ? 'text-blue-600' : 'text-purple-400') : theme.cardSub}`} />
            </div>
          )}
          <span className={`text-sm font-mono truncate ${isSelected ? (isRei ? 'text-blue-700 font-bold' : 'text-purple-300 font-bold') : theme.cardTitle}`}>
            {String(node.name)}
          </span>
        </div>
        {isFile && node.fileData && (
          <div className="flex space-x-2 text-[0.65rem] font-mono shrink-0 ml-2 opacity-80">
            {Number(node.fileData.additions) > 0 && <span className="text-emerald-500">+{node.fileData.additions}</span>}
            {Number(node.fileData.deletions) > 0 && <span className="text-red-500">-{node.fileData.deletions}</span>}
          </div>
        )}
      </div>
      {!isFile && isOpen && (
         <div>
           {Object.values(node.children || {}).map(child => (
             <FileTreeNode key={child.path} node={child} level={level + 1} selectedDiffFile={selectedDiffFile} setSelectedDiffFile={setSelectedDiffFile} theme={theme} isRei={isRei} />
           ))}
         </div>
      )}
    </div>
  );
};

// --- 工具组件：Token 消耗徽章 ---
const TokenUsageBadges = ({ tokens, theme, isRei }) => {
  if (!tokens || Object.keys(tokens).length === 0) return null;
  return (
    <div className="flex flex-wrap gap-2 mt-3">
      {Object.entries(tokens).map(([cli, amount]) => (
        <span key={cli} className={`flex items-center px-1.5 py-0.5 text-[0.65rem] font-mono rounded-sm border ${isRei ? 'bg-indigo-50 text-indigo-600 border-indigo-200' : 'bg-indigo-900/30 text-indigo-400 border-indigo-500/50'}`}>
          <Zap className="w-3 h-3 mr-1 opacity-70" />
          {String(cli)}: {(Number(amount) / 1000).toFixed(1)}k
        </span>
      ))}
    </div>
  );
};

// --- 页面：系统控制台 ---
const ConsolePage = () => {
  const { theme, isRei, projects, tasks, setSelectedProject, setSelectedTask, setProjectView, setActiveMenu, setIsProjectsMenuExpanded } = useApp();

  const activeTasks = tasks.filter(t => t.category === 'ACTIVE');
  
  let totalTokens = 0;
  projects.forEach(p => {
    if (p.tokens?.codex) totalTokens += p.tokens.codex;
    if (p.tokens?.claude) totalTokens += p.tokens.claude;
    if (p.tokens?.gemini) totalTokens += p.tokens.gemini;
  });

  const cliStatus = [
    { id: 'codex-cli', status: '在线', latency: '45ms' },
    { id: 'claude-code', status: '在线', latency: '120ms' },
    { id: 'gemini-cli', status: '离线', latency: '-' }
  ];

  const handleJumpToTask = (task) => {
    const proj = projects.find(p => p.id === task.projectId);
    if (proj) {
      setSelectedProject(proj);
      setSelectedTask(task.id);
      setProjectView('workbench');
      setActiveMenu('projects');
      if (proj.isPinned) {
         setIsProjectsMenuExpanded(true);
      }
    }
  };

  const getStatusBadge = (status) => {
    switch(status) {
      case 'EXECUTING': return <span className={`px-2 py-0.5 border rounded-sm text-xs font-mono font-bold tracking-wider ${theme.badgeExec}`}>[ 执行中 ]</span>;
      case 'PLAN_REVIEW': return <span className={`px-2 py-0.5 border rounded-sm text-xs font-mono font-bold tracking-wider ${theme.badgeWarn.replace('red', 'amber').replace('orange', 'yellow')}`}>[ 计划审阅 ]</span>;
      case 'PAUSED': return <span className={`px-2 py-0.5 border rounded-sm text-xs font-mono tracking-wider ${theme.badgeDraft}`}>[ 已暂停 ]</span>;
      default: return <span className={`px-2 py-0.5 border rounded-sm text-xs font-mono tracking-wider ${theme.badgeDraft}`}>[ {status} ]</span>;
    }
  };

  return (
    <div className="p-8 max-w-7xl mx-auto relative z-10 h-full overflow-y-auto custom-scrollbar flex flex-col space-y-8">
      {/* 精简的顶部监控区域 */}
      <div className={`flex flex-col md:flex-row justify-between md:items-end border-b pb-4 shrink-0 space-y-4 md:space-y-0 ${isRei ? 'border-blue-200/50' : 'border-white/10'}`}>
        <div>
          <div className={`font-mono text-sm tracking-[0.2em] mb-1 ${theme.pageSub}`}>全局概览 //</div>
          <h2 className={`text-3xl font-black tracking-widest ${theme.pageTitle}`}>系统控制台</h2>
        </div>
        
        {/* 聚合状态栏 */}
        <div className="flex flex-col items-end space-y-2 mb-1">
          <div className={`flex space-x-3 text-xs font-mono px-3 py-1.5 rounded-sm border ${isRei ? 'bg-white/50 border-blue-200' : 'bg-black/40 border-purple-500/30'}`}>
             <span className={`flex items-center ${isRei ? 'text-indigo-600' : 'text-indigo-400'}`} title="总消耗">
               <Zap className="w-3.5 h-3.5 mr-1" /> {(totalTokens/1000).toFixed(1)}k
             </span>
             <span className="opacity-30">|</span>
             <span className={`flex items-center ${isRei ? 'text-cyan-600' : 'text-green-400'}`} title="活跃任务数">
               <Activity className="w-3.5 h-3.5 mr-1" /> {activeTasks.length} 个任务
             </span>
             <span className="opacity-30">|</span>
             <span className={`flex items-center ${theme.cardSub}`} title="接管的项目库">
               <FolderGit2 className="w-3.5 h-3.5 mr-1" /> {projects.length} 个项目
             </span>
          </div>
          <div className="flex space-x-4 text-[0.65rem] font-mono">
            {cliStatus.map(cli => (
              <span key={cli.id} className={`flex items-center ${cli.status === '在线' ? (isRei ? 'text-cyan-700' : 'text-green-400') : theme.cardSub}`}>
                <span className={`w-1.5 h-1.5 rounded-full mr-1.5 ${cli.status === '在线' ? (isRei ? 'bg-cyan-500 animate-pulse' : 'bg-green-400 animate-pulse') : 'bg-slate-500'}`}></span>
                {cli.id} ({cli.status === '在线' ? cli.latency : '离线'})
              </span>
            ))}
          </div>
        </div>
      </div>

      {/* 活跃任务板块 */}
      <div className="flex-1">
        <h3 className={`text-lg font-bold font-mono tracking-widest mb-4 flex items-center ${theme.cardTitle}`}>
          <PlayCircle className={`w-5 h-5 mr-2 ${theme.pageSub}`} /> 活跃作战区域
        </h3>
        <div className="grid grid-cols-1 lg:grid-cols-2 xl:grid-cols-3 gap-6">
          {activeTasks.map(task => {
            const project = projects.find(p => p.id === task.projectId) || { name: '未知项目' };
            return (
              <div key={task.id} onClick={() => handleJumpToTask(task)} className={`group relative backdrop-blur-md border p-5 rounded-sm cursor-pointer transition-all duration-300 flex flex-col ${theme.cardBg}`}>
                <div className="flex justify-between items-center mb-3">

                    <FolderGit2 className="w-3 h-3 mr-1" /> {project.name}

                <div className="flex-1 mb-2">
                  <h3 className={`text-lg font-bold mb-1 tracking-wide ${theme.cardTitle}`}>{task.title}</h3>
                  <div className={`text-xs font-mono opacity-60 ${theme.cardSub}`}>ID: {task.id} | 分支: {task.taskBranch}</div>
                </div>

                <TokenUsageBadges tokens={task.tokens} theme={theme} isRei={isRei} />

                <div className={`absolute top-0 left-0 w-2 h-2 border-t-2 border-l-2 transition-colors ${theme.cardCorner}`}></div><div className={`absolute bottom-0 right-0 w-2 h-2 border-b-2 border-r-2 transition-colors ${theme.cardCorner}`}></div>
              </div>
            );
          })}
          {activeTasks.length === 0 && (
            <div className={`col-span-full p-8 text-center rounded-sm border border-dashed font-mono text-sm ${isRei ? 'border-blue-200 text-blue-400 bg-white/50' : 'border-purple-500/30 text-purple-500/70 bg-black/20'}`}>
              当前系统无处于活跃流转态的任务。
            </div>
          )}
        </div>
      </div>
    </div>
  );
};

// --- 其他各种弹窗模块 ---
const RegisterProjectModal = () => {
  const { isRegisterModalOpen, setIsRegisterModalOpen, projects, setProjects, theme, isRei } = useApp();
  const [registerPathInput, setRegisterPathInput] = useState('/Users/dev');
  const [currentBrowsePath, setCurrentBrowsePath] = useState('/Users/dev');
  
  const PRESET_COLORS = ['#eab308', '#f97316', '#ef4444', '#22c55e', '#06b6d4', '#3b82f6', '#a855f7'];
  const [projectColor, setProjectColor] = useState(PRESET_COLORS[0]);

  if (!isRegisterModalOpen) return null;

  const mockFileSystem = {
    '/': [ { name: 'Users', type: 'dir' } ],
    '/Users': [ { name: 'dev', type: 'dir' } ],
    '/Users/dev': [ { name: 'eat-frontend', type: 'dir', isGit: true }, { name: 'backend-api', type: 'dir', isGit: true } ]
  };

  const currentDirs = mockFileSystem[currentBrowsePath] || [];

  const handleDirClick = (dirName) => {
    const newPath = currentBrowsePath === '/' ? `/${dirName}` : `${currentBrowsePath}/${dirName}`;
    setCurrentBrowsePath(newPath);
    setRegisterPathInput(newPath);
  };

  const handleGoUp = () => {
    if (currentBrowsePath === '/') return;
    const parts = currentBrowsePath.split('/');
    parts.pop();
    const newPath = parts.length <= 1 ? '/' : parts.join('/');
    setCurrentBrowsePath(newPath);
    setRegisterPathInput(newPath);
  };

  const handleRegisterSubmit = () => {
    if (!registerPathInput) return;
    const name = registerPathInput.split('/').pop() || 'new-project';
    setProjects([{ id: Math.random().toString(36).substr(2, 5).toUpperCase(), name: name, path: registerPathInput, defaultBranch: 'main', taskCount: 0, status: 'clean', tokens: {}, color: projectColor, isPinned: true }, ...projects]);
    setIsRegisterModalOpen(false);
  };

  return (
    <div className={`fixed inset-0 z-50 flex items-center justify-center transition-all ${theme.modalOverlay}`}>
      <div className={`w-full max-w-2xl p-6 rounded-sm relative overflow-hidden flex flex-col max-h-[90vh] ${theme.modalBox}`}>
         <div className="absolute inset-0 pointer-events-none opacity-10" style={{ backgroundImage: theme.grid, backgroundSize: '20px 20px' }}></div>
         <div className="flex items-center mb-6 relative z-10">
            <div className={`p-2 rounded-sm mr-3 border ${isRei ? 'bg-cyan-50 border-cyan-200' : 'bg-green-900/20 border-green-500/30'}`}>
               <FolderPlus className={`w-5 h-5 ${isRei ? 'text-cyan-600' : 'text-green-400'}`} />
            </div>
            <h3 className={`text-xl font-bold tracking-wider ${isRei ? 'text-cyan-700' : 'text-green-400'}`}>注册本地新项目</h3>
         </div>
         <div className="relative z-10 flex flex-col flex-1 overflow-hidden space-y-4">
            <div>
              <label className={`block text-xs font-mono mb-2 ${theme.cardSub}`}>[ 目标路径 ] 手动输入</label>
              <div className="relative">
                <TerminalSquare className={`absolute left-3 top-1/2 -translate-y-1/2 w-4 h-4 ${isRei ? 'text-blue-400' : 'text-purple-500'}`} />
                <input type="text" value={registerPathInput} onChange={(e) => setRegisterPathInput(e.target.value)} placeholder="输入本地 Git 仓库绝对路径..." className={`w-full pl-9 pr-4 py-2.5 rounded-sm outline-none font-mono text-sm transition-all border ${theme.inputBg}`} />
              </div>
            </div>
            
            <div>
              <label className={`block text-xs font-mono mb-2 ${theme.cardSub}`}>[ 项目颜色 ] 视觉标识</label>
              <div className="flex items-center space-x-3">
                {PRESET_COLORS.map(c => (
                  <button
                    key={c}
                    onClick={() => setProjectColor(c)}
                    className={`w-6 h-6 rounded-full transition-transform ${projectColor === c ? (isRei ? 'scale-125 ring-2 ring-offset-1 ring-blue-400' : 'scale-125 ring-2 ring-offset-1 ring-green-400 ring-offset-[#0a0a0a]') : 'hover:scale-110 opacity-70 hover:opacity-100'}`}
                    style={{ backgroundColor: c }}
                    title={`选择预设颜色 ${c}`}
                  />
                ))}
                <div className={`w-px h-5 mx-2 ${isRei ? 'bg-blue-200' : 'bg-white/20'}`}></div>
                <label className="relative flex items-center justify-center w-7 h-7 rounded-full overflow-hidden border border-slate-400 hover:scale-110 transition-transform cursor-pointer shadow-sm" title="打开调色板选择自定义颜色">
                  <input type="color" value={projectColor} onChange={e => setProjectColor(e.target.value)} className="absolute -top-2 -left-2 w-12 h-12 cursor-pointer opacity-0" />
                  <div className="w-full h-full pointer-events-none" style={{ backgroundColor: projectColor }}></div>
                </label>
                <span className={`text-[0.65rem] font-mono ml-2 opacity-60 uppercase ${theme.cardSub}`}>{projectColor}</span>
              </div>
            </div>

            <div className="flex flex-col flex-1 overflow-hidden border rounded-sm mt-2" style={{ borderColor: isRei ? 'rgba(59,130,246,0.3)' : 'rgba(168,85,247,0.3)' }}>
              <div className={`px-3 py-2 border-b flex items-center text-xs font-mono tracking-wide ${theme.treePathBar}`}>
                <FolderOpen className="w-3.5 h-3.5 mr-2" />
                <span className="truncate flex-1 font-bold">{currentBrowsePath}</span>
              </div>
              <div className={`flex-1 overflow-y-auto custom-scrollbar p-2 ${isRei ? 'bg-white/40' : 'bg-black/20'}`}>
                {currentBrowsePath !== '/' && (
                  <div onClick={handleGoUp} className={`flex items-center px-3 py-2 cursor-pointer transition-colors rounded-sm font-mono text-sm ${theme.treeItemHover} ${theme.cardSub}`}>
                    <CornerLeftUp className="w-4 h-4 mr-3" /><span>.. (返回上级)</span>
                  </div>
                )}
                {currentDirs.length === 0 && (
                  <div className={`px-3 py-4 text-center font-mono text-xs opacity-50 ${theme.cardSub}`}>当前目录下未发现文件夹。</div>
                )}
                {currentDirs.map((dir, idx) => (
                  <div key={idx} onClick={() => handleDirClick(dir.name)} className={`flex items-center justify-between px-3 py-2.5 cursor-pointer transition-colors rounded-sm font-mono text-sm group ${theme.treeItemHover}`}>
                    <div className="flex items-center overflow-hidden">
                      <Folder className={`w-4 h-4 mr-3 shrink-0 ${dir.isGit ? (isRei ? 'text-cyan-500 fill-cyan-500/20' : 'text-green-400 fill-green-400/20') : (isRei ? 'text-blue-400' : 'text-purple-500')}`} />
                      <span className={`truncate ${theme.cardTitle}`}>{dir.name}</span>
                    </div>
                    {dir.isGit && (
                      <div className={`text-[0.6rem] px-1.5 py-0.5 rounded-sm border opacity-70 group-hover:opacity-100 transition-opacity ${isRei ? 'bg-cyan-50 border-cyan-200 text-cyan-600' : 'bg-green-900/30 border-green-500/50 text-green-400'}`}>GIT</div>
                    )}
                  </div>
                ))}
              </div>
            </div>
         </div>
         <div className="flex justify-end space-x-3 mt-6 relative z-10 font-mono text-sm">

const TaskActionModal = () => {
  const { taskActionState, setTaskActionState, deleteBranchChecked, setDeleteBranchChecked, setTasks, theme, isRei } = useApp();

  if (!taskActionState.isOpen || !taskActionState.task) return null;
  const { type, task } = taskActionState;

  const executeTaskAction = () => {
    if (!task) return;
    if (type === 'delete') setTasks(prev => prev.filter(t => t.id !== task.id));
    else if (type === 'archive') setTasks(prev => prev.map(t => t.id === task.id ? { ...t, category: 'ARCHIVED', status: 'COMPLETED' } : t));
    else if (type === 'pause') setTasks(prev => prev.map(t => t.id === task.id ? { ...t, status: task.status === 'PAUSED' ? 'EXECUTING' : 'PAUSED' } : t));
    setTaskActionState({ isOpen: false, type: '', task: null });
  };

  let config = {};
  if (type === 'error') {
    config = { icon: <XCircle className="w-6 h-6 text-red-500" />, title: '操作被拦截：任务处于活跃状态', colorClass: 'text-red-500 border-red-500/30 bg-red-500/10', content: `任务 [${task.id}] 正在被执行或处于审查流中。为保障数据完整性，系统禁止直接归档或删除活跃任务。请先暂停该任务。`, hideCancel: true, confirmText: '收到' };
  } else if (type === 'delete') {
    config = { icon: <AlertTriangle className="w-6 h-6 text-red-500" />, title: '彻底销毁任务记录？', colorClass: 'text-red-500 border-red-500/30 bg-red-500/10', content: `即将从系统中永久抹除任务 [${task.id}]：${task.title} 的所有上下文及快照。此操作不可逆！`, showCheckbox: true, confirmText: '确认销毁' };
  } else if (type === 'archive') {
    config = { icon: <Archive className={`w-6 h-6 ${theme.modalTitleInfo}`} />, title: '将任务转入冷数据归档？', colorClass: `${theme.modalTitleInfo} border-blue-500/30 bg-blue-500/10`, content: `任务 [${task.id}]：${task.title} 将被移出活跃工作台并进入已归档视图。你可以随时在档案库中查阅它的历史。`, confirmText: '确认归档' };
  } else if (type === 'pause') {
    const isPaused = task.status === 'PAUSED';
    config = { icon: isPaused ? <PlayCircle className={`w-6 h-6 ${theme.modalTitleInfo}`} /> : <PauseCircle className={`w-6 h-6 ${theme.modalTitleInfo}`} />, title: isPaused ? '恢复任务执行流？' : '挂起任务执行流？', colorClass: `${theme.modalTitleInfo} border-blue-500/30 bg-blue-500/10`, content: isPaused ? `将重新激活任务 [${task.id}]，对应的 Agent 会话和工作流将继续进行。` : `将中止当前任务 [${task.id}] 的所有 Agent 进程并释放容器资源。你可以随时恢复它。`, confirmText: isPaused ? '确认恢复' : '确认挂起' };
  }
  return (
    <div className={`fixed inset-0 z-50 flex items-center justify-center transition-all ${theme.modalOverlay}`}>
      <div className={`w-full max-w-md p-6 rounded-sm relative overflow-hidden flex flex-col ${theme.modalBox}`}>
         <div className="absolute inset-0 pointer-events-none opacity-10" style={{ backgroundImage: theme.grid, backgroundSize: '20px 20px' }}></div>
         <div className="flex items-start mb-6 relative z-10">
            <div className={`p-3 rounded-full mr-4 border ${config.colorClass}`}>{config.icon}</div>
            <div className="flex-1 mt-1">
              <h3 className={`text-lg font-bold mb-2 tracking-wider ${config.colorClass.split(' ')[0]}`}>{config.title}</h3>
              <p className={`text-sm font-mono leading-relaxed ${theme.cardSub}`}>{config.content}</p>
              {config.showCheckbox && (
                <label className="flex items-center space-x-3 mt-4 cursor-pointer p-2 rounded-sm border border-red-500/20 bg-red-500/5 group">
                  <div className="relative flex items-center justify-center">
                    <input type="checkbox" checked={deleteBranchChecked} onChange={(e) => setDeleteBranchChecked(e.target.checked)} className="peer appearance-none w-4 h-4 border border-red-500/50 rounded-sm checked:bg-red-500 transition-colors cursor-pointer" />
                    <svg className="absolute w-3 h-3 text-white opacity-0 peer-checked:opacity-100 pointer-events-none" fill="none" viewBox="0 0 24 24" stroke="currentColor" strokeWidth="3"><path strokeLinecap="round" strokeLinejoin="round" d="M5 13l4 4L19 7" /></svg>
                  </div>
                  <span className="text-sm font-mono text-red-400 group-hover:text-red-500 transition-colors">同时抹除本地对应分支：<br/><span className="font-bold">{task.taskBranch}</span></span>
                </label>
              )}
            </div>
         </div>
         <div className="flex justify-end space-x-3 relative z-10 font-mono text-sm">
           {!config.hideCancel && (<button onClick={() => setTaskActionState({ isOpen: false, type: '', task: null })} className={`px-4 py-2 rounded-sm border transition-colors ${theme.btnGhost} ${isRei ? 'border-blue-200' : 'border-white/10'}`}>取消</button>)}
           <button onClick={config.hideCancel ? () => setTaskActionState({ isOpen: false, type: '', task: null }) : executeTaskAction} className={`px-4 py-2 rounded-sm border transition-colors flex items-center ${config.hideCancel ? 'bg-red-500/10 border-red-500/50 text-red-500 hover:bg-red-500 hover:text-white' : (type === 'delete' ? 'bg-red-500/10 border-red-500/50 text-red-500 hover:bg-red-500 hover:text-white' : (isRei ? 'bg-blue-50 text-blue-600 border-blue-300 hover:bg-blue-600 hover:text-white' : 'bg-green-900/30 border-green-500/50 text-green-400 hover:bg-green-500 hover:text-black'))}`}>{config.confirmText}</button>
         </div>
      </div>
    </div>
  );
};

        {task.status === 'EXECUTING' && renderExecutingView()}
        {task.status === 'COMPLETED' && renderMergeReviewView()}
      </div>

};
